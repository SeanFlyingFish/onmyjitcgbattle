local zoneGUIDs = {

local MAIN_DECK

-- ====================== 配置（和网页工具完全一致） ======================
local BASE64_CHARS = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/"
local PREFIX_DECODE = { ["1"] = "AYK", ["2"] = "BPP", ["3"] = "GOT" }
local SUFFIX_DECODE = { ["0"] = "", ["1"] = "SP", ["2"] = "OR", ["3"] = "UR", ["4"] = "PC", ["5"] = "EX", ["6"] = "3D" }
local CHAR7_DECODE = { ["0"] = "", ["1"] = "a", ["2"] = "b", ["3"] = "c" }

-- Base64字符获取索引（TTS兼容）
local function getBase64Index(char)
    for i = 1, #BASE64_CHARS do
        if BASE64_CHARS:sub(i,i) == char then
            return i - 1
        end
    end
    return 0
end

-- 自定义Base64解码（复刻网页，纯TTS兼容）
local function customBase64Decode(base64Str)
    local numStr = ""
    for i = 1, #base64Str, 4 do
        local chunk = base64Str:sub(i, i+3)
        if #chunk < 4 then break end
        
        local n1 = getBase64Index(chunk:sub(1,1))
        local n2 = getBase64Index(chunk:sub(2,2))
        local n3 = getBase64Index(chunk:sub(3,3))
        local n4 = getBase64Index(chunk:sub(4,4))
        
        -- 关键：用乘法替代位运算，TTS完美支持
        local num = n1 * 262144 + n2 * 4096 + n3 * 64 + n4
        numStr = numStr .. string.format("%07d", num)
    end
    return numStr
end

-- 单卡牌解码（和网页完全一致）
local function decodeSingleCard(code)
    if #code ~= 7 then return nil end
    local prefix = PREFIX_DECODE[code:sub(1,1)] or ""
    local num = code:sub(2,4)
    local char7 = CHAR7_DECODE[code:sub(5,5)] or ""
    local suffix = SUFFIX_DECODE[code:sub(6,6)] or ""
    local count = tonumber(code:sub(7,7)) or 1

    local cardName = prefix .. num .. char7 .. (suffix ~= "" and "-"..suffix or "")
    return prefix ~= "" and { cardName = cardName, count = count } or nil
end

-- 核心解码函数
function decodeOTCode(encodedStr)
    local cleanStr = encodedStr:gsub("%*OT/", "")
    local fullCode = customBase64Decode(cleanStr)
    local result = {}

    for i = 1, #fullCode, 7 do
        local c = fullCode:sub(i, i+6)
        local card = decodeSingleCard(c)
        if card then
            for _ = 1, card.count do
                table.insert(result, card.cardName)
            end
        end
    end
    return table.concat(result, " ")
end

function setupPlayerZones(playerColor)
    local validColors = {["Red"]=true, ["Blue"]=true, ["Yellow"]=true, ["Green"]=true}
    if not validColors[playerColor] then
        broadcastToAll("请选择 Red, Blue, Yellow, 或 Green", {1, 0.5, 0})
        return false
    end
    MAIN_DECK = zoneGUIDs[string.upper(playerColor) .. "_MAIN_DECK"]
    return true
end

function onSpawnButtonClick(obj, color, clicking_player)
    local playerColor = color
    if not setupPlayerZones(playerColor) then return end

    local inputText = self.getInputs()[1].value:gsub("^%s+", ""):gsub("%s+$", "")

    -- ==========================================
    -- 【仅新增这3行！你的原有代码完全不动】
    local finalText = inputText
    if inputText:match("^%*OT/") then
        finalText = decodeOTCode(inputText) -- 解码
    end
    -- ==========================================

    -- 👇👇👇 以下是你的原版代码，一字未改！👇👇👇
    local cardIDs = {}
    for id in finalText:gmatch("%S+") do table.insert(cardIDs, id:upper()) end

    local spawnPos = {x = 0, y = 2, z = 0}
    for i, cardID in ipairs(cardIDs) do
        local card = cardData[cardID]
        if card then
            Wait.time(function()
                spawnSimpleCard(card, spawnPos, playerColor, cardID)
                spawnPos.y = spawnPos.y + 0.1
            end, i * 0.05)
        end
    end
end

function spawnSimpleCard(card, spawnPosition, playerColor, cardID)
    local zone = getObjectFromGUID(MAIN_DECK)
    if not zone then return end

    local zonePos = zone.getPosition()
    spawnPosition.x = zonePos.x
    spawnPosition.z = zonePos.z

    local isRedSide = (playerColor == "Red" or playerColor == "Yellow")
    local rotation = isRedSide and {0, 180, 180} or {0, 0, 180}

    spawnObject({
        type = "Card",
        position = spawnPosition,
        rotation = rotation,
        scale = {1.5, 1, 1.5},
        callback_function = function(obj)
            obj.setCustomObject({
                face = card.image,
                back = "https://steamusercontent-a.akamaihd.net/ugc/17093582484054773031/E6608BC51B78683BDB6B593B86DEF83528E59434/"
            })
            obj.setName(card.name)
            obj.setDescription(card.description)
            obj.setGMNotes(card.gmNotes)
            if card.tags then
                for _, tag in pairs(card.tags) do
                    obj.addTag(tag)
                end
            end
        end
    })
end
